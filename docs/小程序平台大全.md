## 八、各平台申请流程与审核要求

### 8.1 微信小程序

#### 申请流程
1. 注册微信公众平台账号（https://mp.weixin.qq.com/）
2. 选择"小程序"类型进行注册
3. 完善主体信息（企业/个人）
4. 进行开发者认证（个人免认证，企业需认证）
5. 配置服务器域名
6. 提交代码审核
7. 发布上线

#### 审核要求
| 类别 | 要求 |
|------|------|
| 审核周期 | 1-7个工作日（通常1-3天） |
| 主体认证 | 个人：免认证；企业：300元/年 |
| 类目准入 | 部分类目需资质证明（如金融、医疗） |
| 内容规范 | 不得含诱导分享、虚假宣传等 |
| 隐私政策 | 需提供隐私政策链接 |

#### 注意事项
- 个人小程序功能受限（如无微信支付）
- 部分API需已认证服务号授权
- 涉及用户数据的需有隐私政策

### 8.2 支付宝小程序

#### 申请流程
1. 注册蚂蚁金服开放平台（https://open.alipay.com/）
2. 选择"小程序"类型创建应用
3. 完成企业/个人认证
4. 配置应用信息与域名
5. 提交审核（审核通常1-3天）
6. 发布上线

#### 审核要求
| 类别 | 要求 |
|------|------|
| 审核周期 | 1-3个工作日 |
| 主体认证 | 个人：支持；企业：需认证 |
| 类目准入 | 涉金融类需特殊资质 |
| 性能要求 | 首屏加载≤3秒 |

#### 注意事项
- 支付宝对支付、信用类功能集成更友好
- 支持获取用户公积金、车辆等生活号数据
- 芝麻信用分可直接调用

### 10.1 全局对象适配

```javascript
// platforms/adapter.js - 平台适配器核心

const PLATFORM_WECHAT = 'wechat';
const PLATFORM_ALIPAY = 'alipay';
const PLATFORM_BAIDU = 'baidu';
const PLATFORM_BYTEDANCE = 'bytedance';
const PLATFORM_JD = 'jd';
const PLATFORM_MEITUAN = 'meituan';
const PLATFORM_PINDUODUO = 'pinduoduo';
const PLATFORM_HUAWEI = 'huawei';
const PLATFORM_XIAOMI = 'xiaomi';
const PLATFORM_OPPO = 'oppo';
const PLATFORM_VIVO = 'vivo';

function getPlatform() {
  if (typeof tt !== 'undefined') {
    return PLATFORM_BYTEDANCE;
  } else if (typeof swan !== 'undefined') {
    return PLATFORM_BAIDU;
  } else if (typeof my !== 'undefined') {
    return PLATFORM_ALIPAY;
  } else if (typeof wx !== 'undefined') {
    return PLATFORM_WECHAT;
  } else if (typeof jd !== 'undefined') {
    return PLATFORM_JD;
  } else if (typeof mt !== 'undefined') {
    return PLATFORM_MEITUAN;
  } else if (typeof dd !== 'undefined') {
    return PLATFORM_PINDUODUO;
  } else if (typeof qh !== 'undefined') {
    return PLATFORM_HUAWEI;
  } else if (typeof mi !== 'undefined') {
    return PLATFORM_XIAOMI;
  } else if (typeof qapp !== 'undefined') {
    return PLATFORM_OPPO;
  } else if (typeof qq !== 'undefined') {
    return PLATFORM_VIVO;
  }
  return PLATFORM_WECHAT;
}

function getApi() {
  const platform = getPlatform();
  switch (platform) {
    case PLATFORM_ALIPAY: return my;
    case PLATFORM_BAIDU: return swan;
    case PLATFORM_BYTEDANCE: return tt;
    case PLATFORM_JD: return jd;
    case PLATFORM_MEITUAN: return mt;
    case PLATFORM_PINDUODUO: return dd;
    case PLATFORM_HUAWEI: return qh;
    case PLATFORM_XIAOMI: return mi;
    case PLATFORM_OPPO: return qapp;
    case PLATFORM_VIVO: return qq;
    case PLATFORM_WECHAT:
    default: return wx;
  }
}

module.exports = { getPlatform, getApi };
```

### 10.2 网络请求适配

```javascript
// utils/request.js - 统一网络请求封装

const { getPlatform, getApi } = require('../platforms/adapter.js');

function request(options) {
  const api = getApi();
  const platform = getPlatform();
  const defaultOptions = {
    timeout: 30000,
    header: {
      'Content-Type': 'application/json'
    }
  };

  // 支付宝特殊处理
  if (platform === 'alipay') {
    return new Promise((resolve, reject) => {
      api.request({
        ...defaultOptions,
        ...options,
        success: (res) => resolve(res),
        fail: (err) => reject(err)
      });
    });
  }

  // 微信、百度、抖音统一处理
  return new Promise((resolve, reject) => {
    api.request({
      ...defaultOptions,
      ...options,
      success: (res) => {
        if (res.statusCode >= 200 && res.statusCode < 300) {
          resolve(res.data);
        } else {
          reject(res);
        }
      },
      fail: (err) => reject(err)
    });
  });
}

// GET请求
function get(url, params = {}, options = {}) {
  const queryString = Object.keys(params)
    .map(key => `${encodeURIComponent(key)}=${encodeURIComponent(params[key])}`)
    .join('&');
  const fullUrl = queryString ? `${url}?${queryString}` : url;
  return request({
    url: fullUrl,
    method: 'GET',
    ...options
  });
}

// POST请求
function post(url, data = {}, options = {}) {
  return request({
    url,
    method: 'POST',
    data,
    ...options
  });
}

module.exports = { request, get, post };
```

### 10.3 Toast提示适配

```javascript
// utils/toast.js - 统一Toast提示封装

const { getPlatform, getApi } = require('../platforms/adapter.js');

function showToast(options) {
  const api = getApi();
  const platform = getPlatform();

  // 支付宝Toast参数差异
  if (platform === 'alipay') {
    return api.showToast({
      content: options.title || options.content,
      type: options.icon === 'success' ? 'success' : 'fail',
      duration: options.duration || 2000
    });
  }

  // 微信、百度、抖音统一处理
  return api.showToast({
    title: options.title || options.content,
    icon: options.icon || 'none',
    duration: options.duration || 2000,
    mask: options.mask || false
  });
}

function showLoading(options) {
  const api = getApi();
  const platform = getPlatform();

  if (platform === 'alipay') {
    return api.showLoading({
      content: options.title || '加载中...'
    });
  }

  return api.showLoading({
    title: options.title || '加载中...',
    mask: options.mask !== false
  });
}

function hideLoading() {
  const api = getApi();
  const platform = getPlatform();

  if (platform === 'alipay') {
    return api.hideLoading();
  }

  return api.hideLoading();
}

module.exports = { showToast, showLoading, hideLoading };
```

## 附录C：常见问题与解决方案

### Q1: 各平台登录态如何统一？
**A**: 建立自己的登录态体系，以code换token，各平台获取code的方式封装统一。

### Q2: 支付回调如何处理？
**A**: 后端统一处理各平台支付回调，前端只负责调起支付和查询订单状态。

### Q3: 地图定位坐标不一致怎么办？
**A**: 后端统一做坐标转换，或前端SDK封装统一坐标系（建议统一为GCJ-02）。

### Q4: 样式兼容性问题如何处理？
**A**: 使用rpx单位适配不同屏幕，使用Flex布局，避免平台特有样式。

### Q5: 包体积超限怎么办？
**A**: 分包加载，按需加载，使用Tree-shaking优化代码。
